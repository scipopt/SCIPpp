115 template <typename CoeffType = ConstantCoefficient>

253 template <typename T, typename PT>
254 void setParam(params::Param<PT> parameter, T value)
255 {
256 auto ptValue { static_cast<PT>(value) };
257 const auto* cName { parameter.scipName.data() };
258 if constexpr (std::is_same_v<PT, int>) {
259 m_scipCallWrapper(SCIPsetIntParam(m_scip, cName, ptValue));
260 } else if constexpr (std::is_same_v<PT, double>) {
261 m_scipCallWrapper(SCIPsetRealParam(m_scip, cName, ptValue));
262 } else if constexpr (std::is_same_v<PT, char>) {
263 m_scipCallWrapper(SCIPsetCharParam(m_scip, cName, value));
264 } else if constexpr (std::is_same_v<PT, bool>) {
265 m_scipCallWrapper(SCIPsetBoolParam(m_scip, cName, value));
266 } else if constexpr (std::is_same_v<PT, SCIP_Longint>) {
267 m_scipCallWrapper(SCIPsetLongintParam(m_scip, cName, value));
268 } else if constexpr (std::is_same_v<PT, std::string>) {
269 m_scipCallWrapper(SCIPsetStringParam(m_scip, cName, value.c_str()));
270 } else {
271 // make this not compile
272 // https://www.open-std.org/jtc1/sc22/wg21/docs/papers/2022/p2593r0.html#workarounds
273 static_assert(sizeof(T) == 0);
274 }
275 }
